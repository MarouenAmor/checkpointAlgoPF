
FUNCTION dot_produc(v1,v2:ARRAY_OF FLOAT) : FLOAT
VAR
i:INTEGER;
res:FLOAT;
BEGIN
res:=0;
FOR i FROM 0 TO v1.length-1 STEP 1  DO
res:=res+(v1[i]*v2[i]);
    RETURN res ;
END
